/**
 * @description Representa datos de la persona registrada
 * @class   
 */
class Persona {
    /**
     * 
     * @param {string} nombre 
     * @param {number} edad 
     * @param {string} email 
     */
  constructor(nombre, edad, email) {
    this.nombre = nombre;
    this.edad = edad;
    this.email = email;
  }

  /**
   * @description Saludo  
   * @returns {string} Retorna un saludo utilizando el nombre y edad recibido como parametros
   */
  saludar() {
    return `Hola, soy ${this.nombre} y tengo ${this.edad} años`;
  }
 /**
  * 
  * @param {object} El objeto que se cosntruyó 
  * @returns {object} Una instancia de una persona
  */
  static desdeObjeto(obj) {
    return new Persona(obj.nombre, obj.edad, obj.email);
  }
/**
 * 
 * @param {string} Solo el nombre registrado       
 * @returns {object} Una instacia de una persona con solo el nombre
 */
  static soloNombre(nombre) {
    return new Persona(nombre, null, null);
  }
};
